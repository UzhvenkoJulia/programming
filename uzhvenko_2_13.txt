# Зчитування вхідних даних (довжини сторін трикутника)
a, b, c = map(int, input().split())

# Перевірка типу трикутника
if a == b == c:
    # Рівносторонній трикутник
    print(1)
elif a == b or b == c or a == c:
    # Рівнобедрений трикутник
    print(2)
else:
    # Різносторонній трикутник
    print(3)
